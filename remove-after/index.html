Quickstart TypeScript 
// npm i zentla

import { Zentla } from "zentla";

const z = new Zentla({
  apiKey: process.env.ZENTLA_API_KEY,
});

// Record usage (replay-safe)
await z.usage.record({
  tenantId: "tnt_123",
  metric: "api_calls",
  quantity: 1200,
  idempotencyKey: "evt_2026_01_11_0001",
});

// Entitlements (enforce in-app)
const ent = await z.entitlements.get("tnt_123");
if (!ent.features.advancedReports) throw new Error("Upgrade required");
